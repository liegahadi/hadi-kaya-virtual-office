   362→model Memory {
   363→  id              String   @id @default(cuid())
   364→  agentId         String?  // null = entity memory / umum
   365→  customerId      String?  // customer-specific memory
   366→  category        String   // UTAMA | BERKAS | FINANCE | MATERIAL | MARKETING

   368→  entityType      String?  // CUSTOMER | BANK | OWNER | PROJECT | SUPPLIER
   369→  entityId        String?  // ID of the entity
   370→  title           String?  // Judul memory (like book title)
   371→  content         String   // Deskripsi: what happened / what's the rule
   372→  resolution      String?  // Action items: what to do about it
   373→  importance      Float    @default(0.5)
   374→  source          String?
   375→  attachmentUrl   String?
   376→  version         Int      @default(1)
   377→  isActive        Boolean  @default(true)
   378→  createdAt       DateTime @default(now())
   379→  updatedAt       DateTime @updatedAt
   380→
   381→  agent Agent? @relation(fields: [agentId], references: [id], onDelete: Cascade)
   382→  versions MemoryVersion[]
   383→
   384→  @@index([agentId, category])
   385→  @@index([customerId])
   386→  @@index([memoryType, entityType])
   387→  @@index([isActive])
   388→}
