    43→export interface PendingActionRecord {
    44→  id: string
    45→  conversationId: string | null
    46→  channel: string
    47→  senderNumber: string | null
    48→  type: string  // DELETE | CREATE | UPDATE
    49→  targetType: string  // CUSTOMER | FIELD
    50→  targetId: string | null
    51→  targetName: string | null
    52→  customerData: any | null
    53→  fieldName?: string | null
    54→  newValue?: string | null
    55→}
    56→
    57→const PENDING_TTL_MINUTES = 5
    58→
    59→// Find active pending action for this conversation/sender
    60→// Scoping rules:
    61→//   1. If conversationId provided → scope by conversationId
    62→//   2. Else if channel + senderNumber provided → scope by both (WA per-user)
    63→//   3. Else if channel only provided → scope by channel (DASHBOARD single owner)
    64→export async function getActivePendingAction(opts: {
    65→  conversationId?: string
    66→  channel?: string
    67→  senderNumber?: string
    68→}): Promise<PendingActionRecord | null> {
    69→  try {
    70→    // Expire old pending actions first
    71→    const expiryTime = new Date(Date.now() - PENDING_TTL_MINUTES * 60 * 1000)
    72→    await db.pendingAction.updateMany({
    73→      where: { status: 'PENDING', createdAt: { lt: expiryTime } },
    74→      data: { status: 'EXPIRED' }
    75→    })
    76→
    77→    // Find active one for this context
    78→    const where: any = { status: 'PENDING' }
    79→    if (opts.conversationId) {
    80→      where.conversationId = opts.conversationId
    81→    } else if (opts.channel && opts.senderNumber) {
    82→      where.channel = opts.channel
    83→      where.senderNumber = opts.senderNumber
    84→    } else if (opts.channel) {
    85→      where.channel = opts.channel
    86→      // For DASHBOARD channel, only match where senderNumber is null
    87→      // (so DASHBOARD pending doesn't conflict with WA pending)
    88→      if (opts.channel === 'DASHBOARD') {
    89→        where.OR = [{ senderNumber: null }, { senderNumber: '' }]
    90→      }
    91→    } else {
    92→      // Default to DASHBOARD
    93→      where.channel = 'DASHBOARD'
    94→      where.OR = [{ senderNumber: null }, { senderNumber: '' }]
    95→    }
    96→
    97→    const action = await db.pendingAction.findFirst({
    98→      where,
    99→      orderBy: { createdAt: 'desc' },
   100→    })
   101→    if (!action) return null
   102→    return {
   103→      id: action.id,
   104→      conversationId: action.conversationId,
   105→      channel: action.channel,
   106→      senderNumber: action.senderNumber,
   107→      type: action.type,
   108→      targetType: action.targetType,
   109→      targetId: action.targetId,
   110→      targetName: action.targetName,
   111→      customerData: action.customerData ? JSON.parse(action.customerData) : null,
   112→      fieldName: action.fieldName,
   113→      newValue: action.newValue,
   114→    }
   115→  } catch (err) {
   116→    console.error('getActivePendingAction error:', err)
   117→    return null
   118→  }
   119→}

   121→// Create new pending action (cancels previous pending for same context)
   122→export async function setPendingAction(action: Omit<PendingActionRecord, 'id'>): Promise<void> {
   123→  try {
   124→    // Cancel previous pending for same context (same scoping rules as getActivePendingAction)
   125→    const cancelWhere: any = { status: 'PENDING' }
   126→    if (action.conversationId) {
   127→      cancelWhere.conversationId = action.conversationId
   128→    } else if (action.channel && action.senderNumber) {
   129→      cancelWhere.channel = action.channel
   130→      cancelWhere.senderNumber = action.senderNumber
   131→    } else if (action.channel) {
   132→      cancelWhere.channel = action.channel
   133→      if (action.channel === 'DASHBOARD') {
   134→        cancelWhere.OR = [{ senderNumber: null }, { senderNumber: '' }]
   135→      }
   136→    }
   137→    await db.pendingAction.updateMany({ where: cancelWhere, data: { status: 'CANCELLED' } })
   138→
   139→    // Create new
   140→    await db.pendingAction.create({
   141→      data: {
   142→        conversationId: action.conversationId || null,
   143→        channel: action.channel || 'DASHBOARD',
   144→        senderNumber: action.senderNumber || null,
   145→        type: action.type,
   146→        targetType: action.targetType,
   147→        targetId: action.targetId || null,
   148→        targetName: action.targetName || null,
   149→        customerData: action.customerData ? JSON.stringify(action.customerData) : null,
   150→        fieldName: action.fieldName || null,
   151→        newValue: action.newValue || null,
   152→        status: 'PENDING',
   153→        expiresAt: new Date(Date.now() + PENDING_TTL_MINUTES * 60 * 1000),
   154→      } as any,
   155→    })
   156→  } catch (err) {
   157→    console.error('setPendingAction error:', err)
   158→  }
   159→}

   175→  conversationId?: string
   176→  channel?: string
   177→  senderNumber?: string
   178→}): Promise<void> {
   179→  try {
   180→    const where: any = { status: 'PENDING' }
   181→    if (opts.conversationId) {
   182→      where.conversationId = opts.conversationId
   183→    } else if (opts.channel && opts.senderNumber) {
   184→      where.channel = opts.channel
   185→      where.senderNumber = opts.senderNumber
   186→    } else if (opts.channel) {
   187→      where.channel = opts.channel
   188→      if (opts.channel === 'DASHBOARD') {
   189→        where.OR = [{ senderNumber: null }, { senderNumber: '' }]
   190→      }
   191→    }
   192→    await db.pendingAction.updateMany({ where, data: { status: 'CANCELLED' } })

   194→    console.error('cancelPendingAction error:', err)

   220→export const MEMORY_CATEGORIES = {
   221→  UTAMA: 'UTAMA',         // General knowledge, decisions (DINA)

   223→  FINANCE: 'FINANCE',     // Budget, payments, invoices (RINA)

   228→// Map topic keywords to memory category
   229→export function detectMemoryCategory(message: string): string {
   230→  const msg = message.toLowerCase()
   231→  if (msg.includes('berkas') || msg.includes('dokumen') || msg.includes('kpr') || msg.includes('bank') || msg.includes('ktp') || msg.includes('kk') || msg.includes('npwp') || msg.includes('slip') || msg.includes('sk kerja') || msg.includes('spr') || msg.includes('flpp') || msg.includes('bphtb') || msg.includes('notaris') || msg.includes('konsumen') || msg.includes('debitur')) return MEMORY_CATEGORIES.BERKAS
   232→  if (msg.includes('budget') || msg.includes('anggaran') || msg.includes('pembayaran') || msg.includes('invoice') || msg.includes('rab') || msg.includes('laba') || msg.includes('rugi') || msg.includes('pajak') || msg.includes('dana') || msg.includes('kas')) return MEMORY_CATEGORIES.FINANCE
   233→  if (msg.includes('material') || msg.includes('bahan') || msg.includes('stok') || msg.includes('supplier') || msg.includes('po') || msg.includes('semen') || msg.includes('batu') || msg.includes('pasir')) return MEMORY_CATEGORIES.MATERIAL
   234→  if (msg.includes('marketing') || msg.includes('prospek') || msg.includes('lead') || msg.includes('iklan') || msg.includes('kampanye')) return MEMORY_CATEGORIES.MARKETING
   235→  return MEMORY_CATEGORIES.UTAMA
   236→}

   238→export function detectIntent(message: string): IntentResult {
   239→  const msg = message.toLowerCase()
   240→  const tools: string[] = []
   241→  let action: IntentResult['action'] = 'READ'
   242→  let newBankValue: string | undefined
   243→  let newStageValue: string | undefined
   244→  let intent_updateField: string | undefined
   245→  let intent_updateValue: string | undefined
   246→  let newCustomerName: string | undefined
   247→  let newCustomerPhone: string | undefined
   248→  let newCustomerBlock: string | undefined
   249→  let newCustomerBank: string | undefined
   250→  let intent_requestedFileIndex: number | undefined
   251→  let intent_requestedFileDocType: string | undefined
   252→  let intent_newBankName: string | undefined
   253→  let intent_bankName: string | undefined
   254→  let intent_logoPrompt: string | undefined
   255→
   256→  // === READ INTENTS ===
   257→
   258→  // Customer count / statistics
   259→  if (msg.includes('berapa') && (msg.includes('konsumen') || msg.includes('debitur') || msg.includes('nasabah') || msg.includes('orang'))) {
   260→    tools.push('getCustomerStats')
   261→  }
   262→
   263→  // Customer list / names — broader matching
   264→  if ((msg.includes('siapa') || msg.includes('nama') || msg.includes('list') || msg.includes('daftar') || msg.includes('sebut') || msg.includes('siapa aja') || msg.includes('mana aja')) && (msg.includes('konsumen') || msg.includes('debitur') || msg.includes('nasabah'))) {
   265→    tools.push('getAllCustomers')
   266→  }
   267→
   268→  // Bank distribution
   269→  if ((msg.includes('btn') || msg.includes('mandiri') || msg.includes('bsb')) && (msg.includes('berapa') || msg.includes('distribusi') || msg.includes('masuk') || msg.includes('berapa banyak'))) {
   270→    tools.push('getBankDistribution')
   271→  }
   272→
   273→  // Stage/pipeline
   274→  if (msg.includes('stage') || msg.includes('pipeline') || msg.includes('proses') || msg.includes('tahap') || msg.includes('status')) {
   275→    if (msg.includes('berapa') || msg.includes('distribusi') || msg.includes('ada') || msg.includes('gimana')) {
   276→      tools.push('getStageDistribution')
   277→    }
   278→  }
   279→
   280→  // Dashboard / overview
   281→  if (msg.includes('dashboard') || msg.includes('ringkasan') || msg.includes('overview') || msg.includes('statistik') || msg.includes('summary') || msg.includes('kabar')) {
   282→    tools.push('getAllCustomers')
   283→    tools.push('getStageDistribution')
   284→    tools.push('getBankDistribution')
   285→  }
   286→
   287→  // Document status — broader
   288→  if (msg.includes('berkas') || msg.includes('dokumen') || msg.includes('lengkap') || msg.includes('kurang') || msg.includes('belum') || msg.includes('kurang')) {
   289→    tools.push('getCustomerDocStatus')
   290→  }
   291→
   292→  // Workplace / lokasi
   293→  if ((msg.includes('lokasi') || msg.includes('tempat kerja') || msg.includes('kerja') || msg.includes('maps') || msg.includes('google maps') || msg.includes('alamat kerja')) && !msg.includes('update') && !msg.includes('ubah')) {
   294→    action = 'GET_WORKPLACE'
   295→    tools.push('getAllCustomers')
   296→  }
   297→
   298→  // === WRITE INTENTS (UPDATE) — broader detection ===
   299→
   300→  // Update bank — catch any "ubah/ganti/pindah/update" + bank name
   301→  if ((msg.includes('update') || msg.includes('ubah') || msg.includes('ganti') || msg.includes('pindah') || msg.includes('pindahin') || msg.includes('gantiin') || msg.includes('jadikan') || msg.includes('masukin ke')) && (msg.includes('bank') || msg.includes('btn') || msg.includes('mandiri') || msg.includes('bsb') || msg.includes('syariah'))) {
   302→    action = 'UPDATE_BANK'
   303→    if (msg.includes('btn')) newBankValue = 'BTN'
   304→    if (msg.includes('mandiri')) newBankValue = 'MANDIRI'
   305→    if (msg.includes('bsb') || msg.includes('syariah')) newBankValue = 'BSB_SYARIAH'
   306→    tools.push('getAllCustomers')
   307→  }
   308→
   309→  // Update stage — catch "ubah status jadi sp3k", "ganti stage ke akad", etc
   310→  // Also catch: "jadi sp3k", "jadikan sp3k", "pindah ke sp3k", "maju ke akad"
   311→  const stageKeywords = ['dm', 'survey', 'closing', 'booking', 'slik', 'pemberkasan', 'sp3k', 'sppk', 'akad', 'serah terima', 'serahterima']
   312→  const hasStageKeyword = stageKeywords.some(kw => msg.includes(kw))
   313→  const hasUpdateKeyword = msg.includes('update') || msg.includes('ubah') || msg.includes('ganti') || msg.includes('pindah') || msg.includes('majuin') || msg.includes('pindahin') || msg.includes('jadikan') || msg.includes('jadi') || msg.includes('masuk') || msg.includes('naik')
   314→
   315→  if (hasStageKeyword && (hasUpdateKeyword || msg.includes('status') || msg.includes('stage') || msg.includes('tahap'))) {
   316→    action = 'UPDATE_STAGE'
   317→    const stageMap: Record<string, string> = {
   318→      'dm': 'DM', 'survey': 'SURVEY', 'closing': 'CLOSING', 'booking': 'BOOKING',
   319→      'slik': 'SLIK', 'pemberkasan': 'PEMBERKASAN', 'sp3k': 'SP3K', 'sppk': 'SP3K',
   320→      'akad': 'AKAD', 'serah terima': 'SERAH_TERIMA', 'serahterima': 'SERAH_TERIMA',
   321→    }
   322→    for (const [key, val] of Object.entries(stageMap)) {
   323→      if (msg.includes(key)) { newStageValue = val; break }
   324→    }
   325→    // If stage keyword found but not in map, try uppercase
   326→    if (!newStageValue) {
   327→      for (const kw of stageKeywords) {
   328→        if (msg.includes(kw)) { newStageValue = kw.toUpperCase().replace(' ', '_'); break }
   329→      }
   330→    }
   331→    tools.push('getAllCustomers')
   332→  }
   333→
   334→  // Fill form / isi data
   335→  if ((msg.includes('isi') || msg.includes('fill') || msg.includes('bantu') && msg.includes('data')) && !msg.includes('bank')) {
   336→    action = 'FILL_FORM'
   337→    tools.push('getAllCustomers')
   338→  }
   339→
   340→  // === GET_FILES / SEND_FILE (Berkas request from Google Drive) ===
   341→  // Detect: "minta berkas [konsumen]", "kirim berkas [konsumen]", "berkas [konsumen] apa saja",
   342→  //         "Dina minta KTP [konsumen]", "kirim slip gaji [konsumen]", "list berkas [konsumen]"
   343→  const berkasKeywords = ['berkas', 'file', 'dokumen', 'kirim', 'minta', 'ktp', 'kk', 'npwp', 'slip gaji', 'sk kerja', 'spr', 'flpp', 'bphtb', 'notaris', 'lokasi kerja']
   344→  const hasBerkasIntent = (
   345→    (msg.includes('minta') || msg.includes('kirim') || msg.includes('lihat') || msg.includes('list') || msg.includes('daftar') || msg.includes('cek') || msg.includes('apa saja'))
   346→    && berkasKeywords.some(kw => msg.includes(kw))
   347→  )
   348→
   349→  if (hasBerkasIntent) {
   350→    // Check if user is requesting SPECIFIC file type (e.g., "minta KTP jenni")
   351→    // vs just asking what files are available
   352→    const docTypeKeywords: Array<{ keywords: string[], docType: string }> = [
   353→      { keywords: ['ktp'], docType: 'ktp' },
   354→      { keywords: ['kk', 'kartu keluarga'], docType: 'kk' },
   355→      { keywords: ['npwp'], docType: 'npwp' },
   356→      { keywords: ['slip gaji', 'slip'], docType: 'sk-slip-gaji' },
   357→      { keywords: ['sk kerja', 'surat keterangan kerja'], docType: 'sk-slip-gaji' },
   358→      { keywords: ['spr'], docType: 'spr' },
   359→      { keywords: ['flpp'], docType: 'flpp' },
   360→      { keywords: ['ajb'], docType: 'ajb' },
   361→      { keywords: ['bphtb'], docType: 'bphtb' },
   362→      { keywords: ['notaris'], docType: 'notaris' },
   363→      { keywords: ['lokasi kerja', 'lokasi'], docType: 'lokasi-kerja' },
   364→      { keywords: ['sertifikat', 'shgb'], docType: 'sertifikat' },
   365→    ]
   366→    const matchedDocType = docTypeKeywords.find(dt => dt.keywords.some(kw => msg.includes(kw)))
   367→
   368→    // Check if user picked a number (e.g., "yang nomor 2", "kirim 1 dan 3")
   369→    const numberMatch = msg.match(/(?:nomor|no|urut)?\s*(\d+(?:\s*(?:dan|,|&)\s*\d+)*)/)
   370→    const requestedIndices: number[] = []
   371→    if (numberMatch) {
   372→      // Extract all numbers from the match
   373→      const nums = numberMatch[1].match(/\d+/g)
   374→      if (nums) {
   375→        for (const n of nums) {
   376→          const i = parseInt(n)
   377→          if (i >= 1 && i <= 20) requestedIndices.push(i)  // reasonable range
   378→        }
   379→      }
   380→    }
   381→
   382→    if (matchedDocType || requestedIndices.length > 0) {
   383→      // User wants a specific file → SEND_FILE
   384→      action = 'SEND_FILE'
   385→      if (matchedDocType) intent_requestedFileDocType = matchedDocType.docType
   386→      if (requestedIndices.length > 0) intent_requestedFileIndex = requestedIndices[0]  // First one for now
   387→      tools.push('getCustomerFiles')
   388→      tools.push('getAllCustomers')
   389→    } else {
   390→      // User wants to see what files are available → GET_FILES
   391→      action = 'GET_FILES'
   392→      tools.push('getCustomerFiles')
   393→      tools.push('getAllCustomers')
   394→    }
   395→  }
   396→
   397→  // === GENERIC FIELD UPDATE ===
   398→  // Detect patterns like: "ubah NIK jenni jadi 12345", "ganti alamat andas ke jl xyz", "set penghasilan jenni 5000000"
   399→  if ((msg.includes('ubah') || msg.includes('ganti') || msg.includes('set') || msg.includes('update') || msg.includes('isi'))) {
   400→    const fieldPatterns: Array<{ keywords: string[], field: string }> = [
   401→      { keywords: ['nik', 'ktp', 'no ktp'], field: 'nik' },
   402→      { keywords: ['npwp', 'no npwp'], field: 'npwp' },
   403→      { keywords: ['alamat'], field: 'alamat' },
   404→      { keywords: ['telepon', 'no hp', 'wa', 'whatsapp'], field: 'telepon' },
   405→      { keywords: ['pekerjaan', 'job'], field: 'pekerjaan' },
   406→      { keywords: ['perusahaan', 'nama perusahaan'], field: 'perusahaan' },
   407→      { keywords: ['jabatan', 'posisi'], field: 'jabatan' },
   408→      { keywords: ['penghasilan', 'gaji', 'income'], field: 'penghasilan' },
   409→      { keywords: ['tempat lahir'], field: 'tempat lahir' },
   410→      { keywords: ['tanggal lahir', 'tgl lahir'], field: 'tanggal lahir' },
   411→      { keywords: ['tgl closing', 'tanggal closing'], field: 'tgl closing' },
   412→      { keywords: ['tgl sp3k', 'tanggal sp3k'], field: 'tgl sp3k' },
   413→      { keywords: ['tgl akad', 'tanggal akad'], field: 'tgl akad' },
   414→      { keywords: ['no akad'], field: 'no akad' },
   415→      { keywords: ['atasan', 'nama atasan'], field: 'atasan' },
   416→      { keywords: ['jam operasional'], field: 'jam operasional' },
   417→      { keywords: ['maps link', 'link maps'], field: 'maps link' },
   418→      { keywords: ['short link'], field: 'short link' },
   419→    ]
   420→
   421→    for (const fp of fieldPatterns) {
   422→      if (fp.keywords.some(kw => msg.includes(kw))) {
   423→        // Check this is not just asking about the field (e.g., "berapa penghasilan jenni?")
   424→        const isAsking = msg.includes('berapa') || msg.includes('apa') || msg.includes('siapa')
   425→        if (!isAsking) {
   426→          action = 'UPDATE_FIELD'
   427→          intent_updateField = fp.field
   428→          // Try to extract the value after "jadi" or "ke" or "="
   429→          const valueMatch = msg.match(/(?:jadi|ke|=)\s*(.+?)(?:\s*(?:dong|ya|tolong|saya|untuk|dari|$))/)
   430→          if (valueMatch) intent_updateValue = valueMatch[1].trim()
   431→          tools.push('getAllCustomers')
   432→        }
   433→        break
   434→      }
   435→    }
   436→  }
   437→
   438→  // === CONFIRM PENDING ACTION === (STRICT — v8.2)
   439→  // Only treat as confirmation if message is SHORT and matches confirmation keywords.
   440→  // Long messages with "ya" + action verbs (e.g., "ya hapus aja") are treated as fresh requests, NOT confirmations.
   441→  // This prevents the LLM from confirming the wrong pending action when user actually meant something else.
   442→  //
   443→  // CRITICAL: pendingAction state is now DB-backed. detectIntent can't check it directly
   444→  // (it's synchronous). Instead, we mark the action as 'CONFIRM_DELETE' / 'CONFIRM_CREATE'
   445→  // and executeTools() will check DB for active pending action.
   446→  //
   447→  // We also detect CANCEL_PENDING here: "batal", "cancel", "tidak", "jangan"
   448→  const confirmKeywords = ['ya', 'iya', 'iyaa', 'yes', 'konfirmasi', 'lanjut', 'eksekusi', 'gas', 'oke', 'ok', 'setuju', 'yakin', 'benar', 'lakukan', 'do it', 'sure', 'go']
   449→  const cancelKeywords = ['batal', 'cancel', 'tidak', 'jangan', 'ga jadi', 'gajadi', 'berhenti']
   450→
   451→  const msgTrimmed = msg.trim()
   452→  const isShortMsg = msgTrimmed.length <= 30 // short confirmation phrases only
   453→  const isVeryShort = msgTrimmed.length <= 15 // strict — for confirm/cancel triggers
   454→  // Use word boundary ONLY (no substring) to avoid false positives like "no" in "nomor"
   455→  const isConfirmWord = confirmKeywords.some(kw => {
   456→    const regex = new RegExp(`\\b${kw}\\b`, 'i')
   457→    return regex.test(msgTrimmed)
   458→  })
   459→  const isCancelWord = cancelKeywords.some(kw => {
   460→    const regex = new RegExp(`\\b${kw.replace(/\s+/g, '\\s+')}\\b`, 'i')
   461→    return regex.test(msgTrimmed)
   462→  })
   463→
   464→  if (isCancelWord && isVeryShort) {
   465→    action = 'CANCEL_PENDING'
   466→    tools.push('getAllCustomers')
   467→  } else if (isShortMsg && isConfirmWord) {
   468→    // Only treat as confirmation if message is short (≤30 chars) AND contains confirmation keyword
   469→    // "ya hapus aja" is 13 chars — would match... let's be even stricter:
   470→    // Must be ≤15 chars OR pure confirmation keyword
   471→    const pureConfirm = ['ya', 'iya', 'iyaa', 'yes', 'konfirmasi', 'lanjut', 'eksekusi', 'gas', 'oke', 'ok', 'setuju', 'yakin', 'benar', 'lakukan']
   472→    const isPureConfirm = pureConfirm.some(kw => msgTrimmed.toLowerCase() === kw || msgTrimmed.toLowerCase() === `${kw}.` || msgTrimmed.toLowerCase() === `${kw}!`)
   473→
   474→    if (isPureConfirm || isVeryShort) {
   475→      // Mark as CONFIRM_DELETE or CONFIRM_CREATE — executeTools will check DB
   476→      // and route based on pendingAction.type
   477→      action = 'CONFIRM_DELETE' // generic confirm; executeTools will route to correct type
   478→      tools.push('getAllCustomers')
   479→    }
   480→    // If msg is 16-30 chars and contains confirm word but also other words,
   481→    // DON'T trigger confirmation — let other intent detectors handle it.
   482→  }
   483→
   484→  // === CREATE CUSTOMER ===
   485→  // Detect: "dapat konsumen baru", "tambah konsumen", "daftarin konsumen", "bikin konsumen baru"
   486→  // Also: "namanya [X] di blok [Y]" or "namanya [X] blok [Y]"
   487→  const hasCreateKeyword = msg.includes('tambah') || msg.includes('daftarin') || msg.includes('daftar') || msg.includes('bikin') || msg.includes('buat') || msg.includes('input') || msg.includes('masukin') || msg.includes('dapat konsumen') || msg.includes('konsumen baru')
   488→  const hasCustomerWord = msg.includes('konsumen') || msg.includes('debitur') || msg.includes('nasabah')
   489→
   490→  if (hasCreateKeyword && hasCustomerWord) {
   491→    action = 'CREATE_CUSTOMER'
   492→
   493→    // Extract name: "namanya [X]", "nama [X]", "konsumen [X]", "debitur [X]"
   494→    // CRITICAL: Use \s+ (require whitespace) before keyword in lookahead, NOT \s*
   495→    // Otherwise "Budi" matches "di" inside the word — captures just "Bu" instead of "Budi Santoso"
   496→    const namePatterns = [
   497→      msg.match(/namanya\s+([A-Za-z][A-Za-z\s]+?)(?=\s+(?:di|blok|bank|btn|mandiri|bsb|hp|wa|no|dengan|yang|,|$))/i),
   498→      msg.match(/nama\s+([A-Za-z][A-Za-z\s]+?)(?=\s+(?:di|blok|bank|btn|mandiri|bsb|hp|wa|no|dengan|yang|,|$))/i),
   499→      msg.match(/(?:konsumen|debitur|nasabah)\s+(?:baru\s+)?(?:namanya\s+)?([A-Za-z][A-Za-z\s]+?)(?=\s+(?:di|blok|bank|btn|mandiri|bsb|hp|wa|no|dengan|yang|,|$))/i),
   500→    ]
   501→    for (const m of namePatterns) {
   502→      if (m && m[1]) {
   503→        const stop = ['baru', 'dong', 'ya', 'tolong', 'dengan', 'dari', 'ke', 'yang', 'ini', 'itu', 'namanya', 'nama', 'konsumen', 'debitur', 'nasabah', 'di']
   504→        const words = m[1].trim().split(/\s+/).filter((w: string) => !stop.includes(w.toLowerCase()))
   505→        if (words.length > 0) {
   506→          newCustomerName = words.join(' ')
   507→          break
   508→        }
   509→      }
   510→    }
   511→
   512→    // Extract block: "di blok E7", "blok E7", "di E7", "unit E7"
   513→    const blockPatterns = [
   514→      msg.match(/di\s+blok\s+([A-Za-z]\d+)/i),
   515→      msg.match(/blok\s+([A-Za-z]\d+)/i),
   516→      msg.match(/di\s+([A-Za-z]\d+)/i),
   517→      msg.match(/unit\s+([A-Za-z]\d+)/i),
   518→    ]
   519→    for (const m of blockPatterns) {
   520→      if (m && m[1]) {
   521→        newCustomerBlock = m[1].toUpperCase()
   522→        break
   523→      }
   524→    }
   525→
   526→    // Extract phone
   527→    const phoneMatch = msg.match(/(?:hp|wa|whatsapp|telepon|telp|no)\s*[:.]?\s*(\d{8,15})/) || msg.match(/(\d{10,15})/)
   528→    if (phoneMatch) newCustomerPhone = phoneMatch[1]
   529→
   530→    // Extract bank
   531→    if (msg.includes('btn')) newCustomerBank = 'BTN'
   532→    else if (msg.includes('mandiri')) newCustomerBank = 'MANDIRI'
   533→    else if (msg.includes('bsb') || msg.includes('syariah') || msg.includes('sumsel') || msg.includes('babel')) newCustomerBank = 'BSB_SYARIAH'
   534→
   535→    tools.push('getAllCustomers')
   536→  }
   537→
   538→  // === LOGO GENERATION ===
   539→  // "generate logo [nama perusahaan]" → AI generate logo from prompt
   540→  // "recreate logo [nama perusahaan]" → recreate from uploaded image
   541→  if ((msg.includes('logo') || msg.includes('generate logo') || msg.includes('buat logo') || msg.includes('bikin logo') || msg.includes('recreate logo')) && !msg.includes('logout')) {
   542→    action = 'GENERATE_LOGO' as any
   543→    const logoMatch = message.match(/(?:generate|buat|bikin|recreate)\s+logo\s+(?:untuk\s+|perusahaan\s+|usaha\s+)?(.+)/i)
   544→    if (logoMatch) intent_logoPrompt = logoMatch[1].trim()
   545→    tools.push('getAllCustomers')
   546→  }
   547→
   548→  // === BANK CONFIG MANAGEMENT ===
   549→  // DINA can manage bank configs via chat:
   550→  // "list bank" → list all banks
   551→  // "tambah bank BCA" → create new bank
   552→  // "hapus bank BCA" → delete bank
   553→  if (msg.includes('bank') && (msg.includes('list') || msg.includes('daftar') || msg.includes('berapa'))) {
   554→    if (!msg.includes('konsumen') && !msg.includes('debitur') && !msg.includes('nasabah') && !msg.includes('rekening') && !msg.includes('pindah') && !msg.includes('ubah') && !msg.includes('ganti')) {
   555→      action = 'LIST_BANKS' as any
   556→      tools.push('getAllCustomers')
   557→    }
   558→  }
   559→  if ((msg.includes('tambah') || msg.includes('daftarin') || msg.includes('bikin') || msg.includes('buat') || msg.includes('input')) && msg.includes('bank') && !msg.includes('konsumen')) {
   560→    action = 'ADD_BANK' as any
   561→    // Extract bank code/name
   562→    const bankMatch = message.match(/bank\s+([A-Za-z][A-Za-z\s]+?)(?:\s+(?:baru|dengan|yang|,|$))/i)
   563→    if (bankMatch) intent_newBankName = bankMatch[1].trim()
   564→    tools.push('getAllCustomers')
   565→  }
   566→  // NOTE: DELETE_BANK intent is DISABLED PERMANENTLY.
   567→  // No one can delete banks from the system — not even owner.
   568→  // Banks are permanent records. This is a hard rule.
   569→
   570→  // === DELETE CUSTOMER ===
   571→  // Detect: "hapus konsumen", "hapus jenni", "apus debitur", "delete konsumen", "hilangin konsumen"
   572→  if ((msg.includes('hapus') || msg.includes('apus') || msg.includes('delete') || msg.includes('hilangin') || msg.includes('buang') || msg.includes('remove')) && (msg.includes('konsumen') || msg.includes('debitur') || msg.includes('nasabah'))) {
   573→    action = 'DELETE_CUSTOMER'
   574→    tools.push('getAllCustomers')
   575→  }
   576→
   577→  // Always include memories
   578→  tools.push('getRelevantMemories')
   579→
   580→  // Always include ALL customers as context (so DINA knows everyone)
   581→  if (!tools.includes('getAllCustomers')) {
   582→    tools.push('getAllCustomers')
   583→  }
   584→
   585→  // Extract customer name — case insensitive, try multiple patterns
   586→  let customerName: string | undefined
   587→  // Pattern 1: "konsumen/debitur/nasabah [Name]" (case insensitive)
   588→  const nameMatch1 = message.match(/(?:konsumen|debitur|nasabah)\s+([A-Za-z]+)/i)
   589→  // Pattern 2: "dari/si [Name]"
   590→  const nameMatch2 = message.match(/(?:dari|si)\s+([A-Za-z]+)/i)
   591→  // Pattern 3: any word 3+ chars (lowercase too — will match against DB case-insensitive)
   592→  const nameMatch3 = message.match(/\b([A-Za-z]{3,})\b/)
   593→  // Pattern 4: "ubah/ganti/update [Name] ..."
   594→  const nameMatch4 = message.match(/(?:ubah|ganti|update|pindah|jadikan)\s+(?:status\s+)?(?:konsumen\s+)?(?:debitur\s+)?([A-Za-z]+)/i)
   595→  // Pattern 5: "(minta|kirim|lihat|cek) berkas/dokumen/file [Name]" — consumes both verbs
   596→  const nameMatch5a = message.match(/(?:minta|kirim|lihat|cek|tolong)\s+(?:berkas|dokumen|file)\s+(?:konsumen\s+|debitur\s+|nasabah\s+)?(?:dari\s+|si\s+)?([A-Za-z][A-Za-z\s]+?)(?=\s+(?:yang|di|blok|bank|btn|mandiri|bsb|hp|wa|no|dengan|,|$|\?))/i)
   597→  // Pattern 5b: "berkas/dokumen/file [Name]" — for "berkas Jenni"
   598→  const nameMatch5b = message.match(/(?:berkas|dokumen|file)\s+(?:konsumen\s+|debitur\s+|nasabah\s+)?(?:dari\s+|si\s+)?([A-Za-z][A-Za-z\s]+?)(?=\s+(?:yang|di|blok|bank|btn|mandiri|bsb|hp|wa|no|dengan|,|$|\?))/i)
   599→  // Pattern 6: For "Dina minta berkas Jenni" — name at END of message after "berkas"
   600→  const nameMatch6 = message.match(/(?:berkas|dokumen|file)\s+([A-Za-z]+)$/i)
   601→
   602→  if (nameMatch4) customerName = nameMatch4[1]
   603→  else if (nameMatch1) customerName = nameMatch1[1]
   604→  else if (nameMatch2) customerName = nameMatch2[1]
   605→  else if (nameMatch5a) customerName = nameMatch5a[1]
   606→  else if (nameMatch5b) customerName = nameMatch5b[1]
   607→  else if (nameMatch6) customerName = nameMatch6[1]
   608→  // Don't use nameMatch3 — too noisy (matches "ubah", "jadi", etc)
   609→
   610→  // Filter out common non-name words
   611→  const stopWords = ['ubah', 'ganti', 'update', 'pindah', 'jadikan', 'jadi', 'status', 'stage', 'tahap', 'bank', 'btn', 'mandiri', 'bsb', 'syariah', 'sp3k', 'sppk', 'akad', 'booking', 'closing', 'survey', 'slik', 'pemberkasan', 'serah', 'terima', 'dong', 'ke', 'yang', 'ini', 'itu', 'dari', 'untuk', 'pada', 'dengan', 'agar', 'supaya', 'bisa', 'tolong', 'bantu', 'mohon', 'jenni']
   612→  // Note: 'jenni' is removed from stopwords — we WANT to match it
   613→  const actualStopWords = ['ubah', 'ganti', 'update', 'pindah', 'jadikan', 'jadi', 'status', 'stage', 'tahap', 'bank', 'btn', 'mandiri', 'bsb', 'syariah', 'sp3k', 'sppk', 'akad', 'booking', 'closing', 'survey', 'slik', 'pemberkasan', 'serah', 'terima', 'dong', 'yang', 'ini', 'itu', 'untuk', 'pada', 'dengan', 'agar', 'supaya', 'bisa', 'tolong', 'bantu', 'mohon', 'dengan', 'saya', 'kamu', 'kita']
   614→  if (customerName && actualStopWords.includes(customerName.toLowerCase())) {
   615→    customerName = undefined
   616→  }
   617→
   618→  // Extract block number
   619→  let blockNumber: string | undefined
   620→  const blockMatch = message.match(/blok\s+([A-Za-z]\d+)/i) || message.match(/\b([A-Za-z]\d{1,3})\b/)
   621→  if (blockMatch && blockMatch[1].length <= 4) blockNumber = blockMatch[1].toUpperCase()
   622→
   623→  return { tools: [...new Set(tools)], customerName, blockNumber, action, newBankValue, newStageValue, updateField: intent_updateField, updateValue: intent_updateValue, newCustomerName, newCustomerPhone, newCustomerBlock, newCustomerBank, requestedFileIndex: intent_requestedFileIndex, requestedFileDocType: intent_requestedFileDocType, newBankName: intent_newBankName, bankName: intent_bankName, logoPrompt: intent_logoPrompt }
   624→}

   748→const REQUIRED_DOCS = [
   749→  { id: 'ktp', label: 'KTP' }, { id: 'kk', label: 'KK' }, { id: 'npwp', label: 'NPWP' },
   750→  { id: 'status-nikah', label: 'Akta Nikah/Belum Menikah' }, { id: 'slip-gaji', label: 'Slip Gaji' },

   756→  { id: 'aplikasi-signed', label: 'Form Aplikasi (TTD)' }, { id: 'pernyataan-penghasilan-signed', label: 'Surat Pernyataan Penghasilan (TTD)' },

   759→]
   760→
   761→async function getCustomerDocStatus(customerId?: string, customerName?: string): Promise<ToolResult> {
   762→  let customer: any = null
   763→  if (customerId) {
   764→    customer = await db.customer.findUnique({ where: { id: customerId }, include: { units: true } })
   765→  } else if (customerName) {
   766→    customer = await db.customer.findFirst({ where: { name: { contains: customerName, mode: 'insensitive' } }, include: { units: true } })
   767→  } else {
   768→    // Get all customers' doc status
   769→    const allCustomers = await db.customer.findMany({ include: { units: true }, take: 20 })
   770→    const summaries = allCustomers.map(c => {
   771→      const uploadedDocs = c.uploadedDocs ? JSON.parse(c.uploadedDocs) : {}
   772→      const uploadedIds = Object.keys(uploadedDocs)
   773→      const missing = REQUIRED_DOCS.filter(d => !uploadedIds.includes(d.id)).map(d => d.label)
   774→      const block = c.units?.[0]?.blockNumber || (c.blockLetter || '') + (c.houseNumber || '') || '-'
   775→      return `- ${c.name} (Blok ${block}): ${uploadedIds.length} dokumen, BELUM: ${missing.join(', ') || 'semua lengkap'}`
   776→    }).join('\n')
   777→    return { toolName: 'getCustomerDocStatus', success: true, data: allCustomers, summary: `Status berkas semua konsumen:\n${summaries}` }
   778→  }
   779→
   780→  if (!customer) return { toolName: 'getCustomerDocStatus', success: false, data: null, summary: 'Konsumen tidak ditemukan.' }
   781→
   782→  const uploadedDocs = customer.uploadedDocs ? JSON.parse(customer.uploadedDocs) : {}
   783→  const uploadedIds = Object.keys(uploadedDocs)
   784→  const requiredMissing = REQUIRED_DOCS.filter(d => !uploadedIds.includes(d.id))
   785→  const signedMissing = SIGNED_DOCS.filter(d => !uploadedIds.includes(d.id))
   786→  const block = customer.units?.[0]?.blockNumber || (customer.blockLetter || '') + (customer.houseNumber || '') || '-'
   787→
   788→  return {
   789→    toolName: 'getCustomerDocStatus',
   790→    success: true,
   791→    data: { customer, requiredMissing, signedMissing },
   792→    summary: `Konsumen: ${customer.name} (Blok ${block}, Bank: ${customer.bankName || 'belum'}, Stage: ${customer.stage})
   793→Dokumen wajib SUDAH: ${REQUIRED_DOCS.filter(d => uploadedIds.includes(d.id)).map(d => d.label).join(', ') || 'tidak ada'}
   794→Dokumen wajib BELUM: ${requiredMissing.map(d => d.label).join(', ') || 'semua lengkap'}
   795→Dokumen TTD BELUM: ${signedMissing.map(d => d.label).join(', ') || 'semua lengkap'}

   802→// ============================================================
   803→// HELPER: Find customer by name with duplicate detection
   804→// Returns:
   805→//   - { customer, duplicates: [] } if exactly 1 match
   806→//   - { customer: first, duplicates: [...] } if 2+ matches (caller must disambiguate)
   807→//   - null if no match
   808→//
   809→// CRITICAL: 2+ customers can have SAME exact name + birthdate (rare but legal).
   810→// When this happens, DINA must NOT silently pick the first one — that causes
   811→// wrong-customer operations (e.g., delete wrong person, send wrong berkas).
   812→// Instead: list all matches + ask user to disambiguate by blok or NIK.
   813→// ============================================================
   814→
   815→export interface CustomerLookupResult {
   816→  customer: any  // first match (for backwards-compat in non-critical paths)
   817→  duplicates: any[]  // empty if only 1 match, otherwise ALL matches (including customer)
   818→  needsDisambiguation: boolean
   819→}
   820→
   821→export async function findCustomerWithDisambiguation(
   822→  name: string,
   823→  options?: { nik?: string; blockNumber?: string }
   824→): Promise<CustomerLookupResult | null> {
   825→  // If NIK provided, use that (NIK is unique per person by Indonesian law)
   826→  if (options?.nik) {
   827→    const byNik = await db.customer.findFirst({
   828→      where: { nik: options.nik },
   829→      include: { units: true },
   830→    })
   831→    if (byNik) {
   832→      return { customer: byNik, duplicates: [], needsDisambiguation: false }
   833→    }
   834→    // NIK not found — fall through to name search
   835→  }
   836→
   837→  // If blockNumber provided, use that (each unit is unique)
   838→  if (options?.blockNumber) {
   839→    const byBlock = await db.customer.findFirst({
   840→      where: {
   841→        OR: [
   842→          { blockLetter: { startsWith: options.blockNumber[0], mode: 'insensitive' }, houseNumber: { contains: options.blockNumber.substring(1), mode: 'insensitive' } },
   843→          { units: { some: { blockNumber: { equals: options.blockNumber, mode: 'insensitive' } } } }
   844→        ]
   845→      },
   846→      include: { units: true },
   847→    })
   848→    if (byBlock) {
   849→      return { customer: byBlock, duplicates: [], needsDisambiguation: false }
   850→    }
   851→  }
   852→
   853→  // Try exact name match first (case-insensitive)
   854→  let matches = await db.customer.findMany({
   855→    where: { name: { equals: name, mode: 'insensitive' } },
   856→    include: { units: true },
   857→  })
   858→
   859→  // If no exact match, fall back to contains
   860→  if (matches.length === 0) {
   861→    matches = await db.customer.findMany({
   862→      where: { name: { contains: name, mode: 'insensitive' } },
   863→      include: { units: true },
   864→    })
   865→  }
   866→
   867→  if (matches.length === 0) return null
   868→  if (matches.length === 1) return { customer: matches[0], duplicates: [], needsDisambiguation: false }
   869→
   870→  // Multiple matches — need disambiguation
   871→  return {
   872→    customer: matches[0],  // first match (caller can choose to use or not)
   873→    duplicates: matches,
   874→    needsDisambiguation: true,
   875→  }
   876→}

   878→// Build disambiguation message for user
   879→export function buildDisambiguationMessage(duplicates: any[]): string {
   880→  const list = duplicates.map((c: any, i: number) => {
   881→    const block = (c.blockLetter || '') + (c.houseNumber || '') || '-'
   882→    const bank = c.bankName || 'belum dipilih'
   883→    const stage = c.stage || '-'
   884→    const nikSuffix = c.nik ? ` (NIK: ...${c.nik.substring(c.nik.length - 4)})` : ''
   885→    const phone = c.whatsappNumber ? ` | WA: ${c.whatsappNumber}` : ''
   886→    return `${i + 1}. **${c.name}** — Blok ${block} | Bank ${bank} | Stage ${stage}${nikSuffix}${phone}`
   887→  }).join('\n')
   888→
   889→  return `⚠️ **Ditemukan ${duplicates.length} konsumen dengan nama yang cocok.**
   890→
   891→${list}
   892→
   893→Sebutkan **blok** atau **NIK** untuk memastikan konsumen yang mana. Contoh:
   894→• "yang blok E5"
   895→• "yang NIK 1234567890"
   896→• "yang bank BTN"`
   897→}

  1011→async function createCustomer(data: { name: string; phone?: string; block?: string; bank?: string }): Promise<ToolResult> {
  1012→  try {

  1021→    if (data.block) {
  1022→      const match = data.block.match(/^([A-Za-z]+)(\d+.*)$/)
  1023→      if (match) { blockLetter = match[1]; houseNumber = match[2] }
  1024→      else { blockLetter = data.block }
  1025→    }
  1026→

  1028→    if (data.block) {

  1031→          name: { equals: data.name, mode: 'insensitive' },
  1032→          blockLetter: blockLetter,
  1033→          houseNumber: houseNumber,
  1034→        }

  1036→      if (existing) {
  1037→        return {
  1038→          toolName: 'createCustomer',
  1039→          success: false,
  1040→          data: existing,

  1042→        }
  1043→      }
  1044→    }
  1045→
  1046→    const customer = await db.customer.create({
  1047→      data: {
  1048→        name: data.name,
  1049→        whatsappNumber: data.phone || null,
  1050→        phone: data.phone || null,
  1051→        bankName: data.bank || null,
  1052→        blockLetter: blockLetter || null,
  1053→        houseNumber: houseNumber || null,
  1054→        stage: 'DM',

  1056→        closingDate: new Date(),
  1057→      } as any,
  1058→    })
  1059→

  1061→    // (Unit creation requires landSize/buildingSize/price/dpAmount which we don't have here.
  1062→    //  Those values come from site plan data, not from chat input.)
  1063→    let unitLinked = false

  1066→        where: { blockNumber: { equals: data.block, mode: 'insensitive' }, projectId: project.id }

  1070→        await db.unit.update({
  1071→          where: { id: existingUnit.id },
  1072→          data: { customerId: customer.id, status: 'BOOKED', bookedAt: new Date() },
  1073→        })
  1074→        unitLinked = true
  1075→      }
  1076→      // If no existing unit found, that's OK — owner can assign manually later via dashboard
  1077→    }
  1078→
  1079→    return {
  1080→      toolName: 'createCustomer',
  1081→      success: true,
  1082→      data: customer,
  1083→      summary: `✅ Berhasil menambahkan konsumen baru: ${customer.name}${data.block ? ` (Blok ${data.block})` : ''}${data.bank ? ` Bank: ${data.bank}` : ''}${data.phone ? ` Telp: ${data.phone}` : ''}. Stage awal: DM. Data tersimpan di database.${unitLinked ? ' Unit berhasil di-link.' : (data.block ? ' Unit belum di-link (blok tidak ditemukan di site plan, owner bisa assign manual).' : '')}`,
  1084→    }
  1085→  } catch (err: any) {
  1086→    return { toolName: 'createCustomer', success: false, data: null, summary: `❌ GAGAL menambahkan konsumen: ${err?.message || 'unknown error'}` }
  1087→  }

  1113→    //   → when customer is deleted, GoogleDoc records have customerId set to null automatically
  1114→    //   → Drive files remain accessible (we never call Drive API to delete)
  1115→    // - We do NOT call db.googleDoc.deleteMany anymore — that would erase DB metadata

  1118→    //   (we don't want to mark a unit as AVAILABLE just because customer was deleted from DB)

  1121→    await db.unit.updateMany({

  1123→      data: { customerId: null, status: 'AVAILABLE', bookedAt: null, soldAt: null },
  1124→    })

  1126→    // Delete conversation history (chat logs are not Drive files — safe to delete)

  1128→
  1129→    // Delete stage history (internal tracking, not user-facing)
  1130→    await db.customerStageHistory.deleteMany({ where: { customerId } })
  1131→
  1132→    // Delete bank pipelines (internal tracking)
  1133→    await db.bankPipeline.deleteMany({ where: { customerId: customerId as any } })
  1134→
  1135→    // === Finally delete the customer (GoogleDoc records auto-set customerId=null via SetNull) ===
  1136→    await db.customer.delete({ where: { id: customerId } })
  1137→
  1138→    return {
  1139→      toolName: 'deleteCustomer',
  1140→      success: true,
  1141→      data: {
  1142→        deletedCustomer: customer.name,

  1145→        googleDocsPreserved: googleDocCount,

  1147→      summary: `✅ Berhasil MENGHAPUS konsumen: ${customer.name} (Blok ${block}, Bank: ${bank}) dari database.

  1149→📦 **Google Drive files DIPERTAHANKAN** (${googleDocCount} file metadata tetap tersimpan di DB dengan customerId=NULL). File di Drive tidak dihapus dan masih bisa diakses jika konsumen ini didaftarkan ulang di kemudian hari.
  1150→
  1151→🏠 Unit site plan di-unlink (status kembali AVAILABLE), tidak dihapus dari site plan.`,

  1921→export function extractLearning(userMessage: string, aiResponse: string): { content: string; category: string; importance: number } | null {
  1922→  const msg = userMessage.toLowerCase()
  1923→  const category = detectMemoryCategory(userMessage)
  1924→
  1925→  if (msg.includes('update') || msg.includes('ubah') || msg.includes('ganti') || msg.includes('pindah')) {
  1926→    return { content: `User melakukan UPDATE: "${userMessage.substring(0, 200)}". DINA executed action.`, category: 'DECISION', importance: 0.8 }
  1927→  }
  1928→  if (msg.includes('syarat') || msg.includes('requirement') || msg.includes('proses') || msg.includes('tahap')) {
  1929→    return { content: `User bertanya: "${userMessage.substring(0, 200)}". Jawaban: ${aiResponse.substring(0, 300)}`, category, importance: 0.7 }
  1930→  }
  1931→  if (msg.includes('konsumen') || msg.includes('debitur') || msg.includes('berkas') || msg.includes('bank') || msg.includes('kpr')) {
  1932→    return { content: `User bertanya: "${userMessage.substring(0, 200)}"`, category, importance: 0.4 }
  1933→  }
  1934→  if (category !== MEMORY_CATEGORIES.UTAMA) {
  1935→    return { content: `User bertanya soal ${category}: "${userMessage.substring(0, 200)}"`, category, importance: 0.4 }
  1936→  }
  1937→
  1938→  return null
  1939→}
